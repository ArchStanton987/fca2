export type CombatId = string;

export interface DbCombat {
  info: {
    gameId: string;
    date: string; // ISO string
    location?: string;
    title: string;
    description: string;
    contenders: Record<PlayableId, PlayableId>;
  };
  history: Record<number, Record<number, Action>>;
  state: {
    action: Action;
    actorIdOverride: string;
  };
}

export type CombatInfo = DbCombat["info"] & { combatId: CombatId };

export type CombatHistory = DbCombat["history"];

export type CombatState = DbCombat["state"];
